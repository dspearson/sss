<objective>
Extend the data model and version-dispatch layer so that:
1. `UserConfig` gains an optional `hybrid_public` field that round-trips through `.sss.toml` with zero breakage for existing v1 repos.
2. `sss users add-hybrid-key <user> <b64>` writes a user's hybrid public key into the project config.
3. `resolve_suite_from_version` accepts `"2.0"` (currently it errors) and `load_project_config_internal` dispatches `open_repo_key` through `suite_for` so v2 repos are loadable.

Purpose: The `sss migrate` command (Plan 02) depends on all three of these changes. Without this wave, there is nowhere to store per-user hybrid public keys in `.sss.toml` and no way to load a v2 repo after migration.
Output: Modified src/project.rs, src/config.rs, src/commands/users.rs, src/commands/mod.rs, src/main.rs.
</objective>

  ```rust
  // Determine which suite to use from the loaded config's version field.
  // For v1 repos this resolves to ClassicSuite; for v2 (hybrid) it resolves
  // to HybridCryptoSuite. suite_for errors loudly if the hybrid feature is

  let open_suite = suite_for(config.suite()?)?;

  // For v2 repos, the user_keypair must be a KeyPair::Hybrid.
  // Load from keystore when the config is v2; use already-matched keypair for v1.
  #[cfg(feature = "hybrid")]
  let user_keypair = if config.suite()? == crate::crypto::Suite::Hybrid {
      let current_id = keystore.get_current_key_id()?;
      let password = /* re-use the password already obtained above, or prompt again */;
      let hybrid_kp = keystore.load_hybrid_keypair(&current_id, password.as_deref())?;

      user_keypair
  };
  ```

     ```rust
     #[cfg(feature = "hybrid")]
     fn handle_users_add_hybrid_key(sub_matches: &ArgMatches) -> Result<()> {
         use base64::Engine as _;
         use crate::constants::HYBRID_PUBLIC_KEY_SIZE;

         let username = sub_matches.get_one::<String>("username").unwrap();
         let hybrid_b64 = sub_matches.get_one::<String>("hybrid-pubkey").unwrap();

         // Validate length before touching disk
         let raw = base64::prelude::BASE64_STANDARD
             .decode(hybrid_b64)
             .map_err(|e| anyhow!("invalid base64 for hybrid public key: {e}"))?;
         if raw.len() != HYBRID_PUBLIC_KEY_SIZE {
             return Err(anyhow!(
                 "hybrid public key must be {HYBRID_PUBLIC_KEY_SIZE} bytes when decoded, got {}",
                 raw.len()
             ));
         }

         let config_path = crate::config::get_project_config_path()?;
         let mut config = ProjectConfig::load_from_file(&config_path)
             .map_err(|_| anyhow!("No project configuration found. Run 'sss init' first."))?;

         let user = config.users.get_mut(username.as_str()).ok_or_else(|| {
             anyhow!("User '{}' not found in project", username)
         })?;
         user.hybrid_public = Some(hybrid_b64.clone());

         config.save_to_file(&config_path)?;
         println!("Registered hybrid public key for user '{username}'");
         Ok(())
     }
     ```
